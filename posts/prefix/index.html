一维前缀和

# template
# original arr: [1, 2, 3, 4, 5]  n
# prefix sum  : [0, 1, 3, 6, 10, 15] n+1
def get_prefix_sum(self, arr):
    prefix_sum = [0]
    for i in range(len(arr)):
        prefix_sum.append(prefix_sum[-1]+arr[i])
    return prefix_sum

def get_prefix_sum(self, matrix):
    n, m = len(matrix), len(matrix[0])
    prefix_sum = [[0]*(m+1) for _ in range(n+1)]
    for i in range(1, n+1):
        for j in range(1, m+1):
            prefix_sum[i][j] = prefix_sum[i-1][j] 
            					+ prefix_sum[i][j-1]
            					- prefix_sum[i-1][j-1]
                    			+ matrix[i-1][j-1]
     return prefix_sum

Prefix Minimum & Maximum

149 · Best Time to Buy and Sell Stock

from typing import (
    List,
)

class Solution:
    """
    @param prices: Given an integer array
    @return: Maximum profit
    """
    def max_profit(self, prices: List[int]) -> int:
        # write your code here
        min_cost = prices[0]
        max_price = prices[0]
        profit = 0
        for price in prices[1:]:
            if price > max_price:
                max_price = price
                profit = max(profit, max_price-min_cost)
            if price < min_cost:
                max_price = min_cost = price
        return profit
